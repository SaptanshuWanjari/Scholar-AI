# Auto-Save Draft Storage Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add change-based auto-save with draft flags for notebooks and in-progress quiz sessions, with a restore banner on reopen.

**Architecture:** Backend gains `is_draft` on `Notebook` and three session columns on `SavedQuiz`, with two new quiz session endpoints. Frontend adds a 2-second debounce hook to notebook saves and a module-level debounce in the quiz store for session PATCHes.

**Tech Stack:** FastAPI + SQLAlchemy (SQLite), React + Zustand, TypeScript

## Global Constraints

- All new SQLAlchemy columns must also be added to `_ADDED_COLUMNS` in `scholarcli/storage/__init__.py` for live DB migration.
- Backend response field names are snake_case; TypeScript interfaces mirror them exactly.
- `json()` helper in `api.ts` defaults to `method: "POST"` — use `{ ...json(body), method: "PATCH" }` for PATCH calls.
- Tests use `conftest.py`'s `clean_config` fixture which creates a fresh temp DB; import `init_db` and call it before any DB operations.

@router.patch("/quizzes/{quiz_id}/session", response_model=QuizOut)
def patch_quiz_session(quiz_id: int, payload: QuizSessionPatch) -> QuizOut:
    session = get_session()
    try:
        quiz = session.get(SavedQuiz, quiz_id)
        if not quiz:
            raise HTTPException(status_code=404, detail="Quiz not found")
        quiz.session_answers = payload.session_answers
        quiz.session_current_question = payload.session_current_question
        if quiz.session_started_at is None:
            quiz.session_started_at = datetime.now(timezone.utc)
        session.commit()
        session.refresh(quiz)
        return _quiz_out(quiz)
    finally:
        session.close()


@router.delete("/quizzes/{quiz_id}/session", status_code=204)
def clear_quiz_session(quiz_id: int) -> None:
    session = get_session()
    try:
        quiz = session.get(SavedQuiz, quiz_id)
        if not quiz:
            raise HTTPException(status_code=404, detail="Quiz not found")
        quiz.session_answers = None
        quiz.session_current_question = None
        quiz.session_started_at = None
        session.commit()
    finally:
        session.close()

- [ ] **Step 1: Create `frontend/src/app/hooks/useAutoSave.ts`**

```ts
import { useCallback, useEffect, useRef, useState } from "react";

export function useAutoSave<T>(
  saveFn: (data: T) => Promise<void>,
  delay = 2000,
) {
  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
  const pendingRef = useRef<T | null>(null);
  const [saving, setSaving] = useState(false);
  const [lastSaved, setLastSaved] = useState<string>("");

  const schedule = useCallback(
    (data: T) => {
      pendingRef.current = data;
      if (timerRef.current) clearTimeout(timerRef.current);
      timerRef.current = setTimeout(async () => {
        setSaving(true);
        try {
          await saveFn(data);
          setLastSaved("just now");
        } finally {
          setSaving(false);
        }
      }, delay);
    },
    [saveFn, delay],
  );

  const flush = useCallback(() => {
    if (timerRef.current && pendingRef.current !== null) {
      clearTimeout(timerRef.current);
      timerRef.current = null;
      saveFn(pendingRef.current).catch(() => {});
      pendingRef.current = null;
    }
  }, [saveFn]);

  useEffect(() => {
    return () => {
      flush();
    };
  }, [flush]);

  return { schedule, flush, saving, lastSaved };
}
```
